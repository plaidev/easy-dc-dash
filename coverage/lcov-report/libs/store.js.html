  content.forEach((record) => {

    if ((item instanceof Array) || typeof item === 'array') {
      item.forEach((i) => {
        if (!(i.key in labels)) labels[i.key] = i.label
      })
    }
    else {
      if (!(item.key in labels)) labels[item.key] = item.label
    }
  })
